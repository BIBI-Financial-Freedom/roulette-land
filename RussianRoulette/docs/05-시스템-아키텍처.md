## 5. 핵심 엔진 설계

### 5.1 ChamberEngine

```javascript
// ChamberEngine.js
export class ChamberEngine {
  #chamber;       // boolean[6] — private
  #position;      // number — private
  #revealed;      // boolean[6] — 공개 여부

  /**
   * 탄창 생성
   * @param {number} bulletCount - 탄환 수 (1~3)
   */
  create(bulletCount = 1) {
    this.#chamber = Array(6).fill(false);
    this.#revealed = Array(6).fill(false);
    // bulletCount개의 랜덤 위치에 탄환 배치
    const positions = RNG.sample([0,1,2,3,4,5], bulletCount);
    positions.forEach(pos => this.#chamber[pos] = true);
  }

  /** 탄창 스핀 — 시작 위치 무작위화 */
  spin() {
    this.#position = RNG.getInt(0, 5);
    this.#revealed = Array(6).fill(false);
  }

  /** 방아쇠 당기기 — 현재 칸 결과 반환 */
  pull() {
    const result = this.#chamber[this.#position];
    this.#revealed[this.#position] = true;
    this.#position = (this.#position + 1) % 6;
    return result; // true = 꽝, false = 생존
  }

  /** UI에 노출 가능한 상태만 반환 (탄환 위치 숨김) */
  getVisibleState() {
    return this.#revealed.map((revealed, i) => {
      if (!revealed) return 'unknown';
      return this.#chamber[i] ? 'bullet' : 'empty';
    });
  }
}
```

### 5.2 RNG (Random Number Generator)

```javascript
// infra/RNG.js
export const RNG = {
  /** 암호학적 안전 정수 난수 (min~max 포함) */
  getInt(min, max) {
    const range = max - min + 1;
    const array = new Uint32Array(1);
    crypto.getRandomValues(array);
    return min + (array[0] % range);
  },

  /** 배열에서 n개 무작위 추출 (비복원) */
  sample(array, n) {
    const shuffled = [...array];
    for (let i = shuffled.length - 1; i > 0; i--) {
      const j = this.getInt(0, i);
      [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
    }
    return shuffled.slice(0, n);
  },

  /** Fisher-Yates 셔플 */
  shuffle(array) {
    return this.sample(array, array.length);
  }
};
```

### 5.3 TurnEngine

```javascript
// engine/TurnEngine.js
export class TurnEngine {
  #players;       // string[]
  #currentIndex;  // number
  #eliminated;    // Set<string> (서바이벌 모드용)

  constructor(players) {
    this.#players = [...players];
    this.#currentIndex = 0;
    this.#eliminated = new Set();
  }

  getCurrentPlayer() {
    return this.#players[this.#currentIndex];
  }

  nextTurn() {
    do {
      this.#currentIndex = (this.#currentIndex + 1) % this.#players.length;
    } while (this.#eliminated.has(this.#players[this.#currentIndex]));
  }

  eliminate(player) {
    this.#eliminated.add(player);
  }

  getActivePlayers() {
    return this.#players.filter(p => !this.#eliminated.has(p));
  }

  reset() {
    this.#currentIndex = 0;
    this.#eliminated.clear();
  }
}
```

### 6.3 Observer 패턴 (간이 구현)

```javascript
// state/GameState.js
class GameState {
  #state;
  #listeners;

  constructor() {
    this.#state = { ...initialState };
    this.#listeners = new Set();
  }

  getState() {
    return { ...this.#state }; // 읽기 전용 복사본
  }

  setState(partial) {
    this.#state = { ...this.#state, ...partial };
    this.#notify();
  }

  subscribe(listener) {
    this.#listeners.add(listener);
    return () => this.#listeners.delete(listener);
  }

  #notify() {
    this.#listeners.forEach(fn => fn(this.getState()));
  }

  reset() {
    this.#state = { ...initialState };
    this.#notify();
  }
}

export const gameState = new GameState();
```

```
[사용자 입력]
     │
     ▼

     │
     ▼

     │
     ▼

     │
     ▼
[Observer Notify] ── 등록된 리스너 실행
     │
     ├──► [UI Update] ── 화면 갱신
     └──► [Effects] ── 사운드, 애니메이션 트리거
```

### 7.2 핵심 이벤트 시퀀스 — 방아쇠 당기기

```
1. User: TriggerButton.click
2. GameScreen: 버튼 비활성화
3. ChamberEngine: pull() → result (true/false)
4. JudgeEngine: judge(result) → verdict
5. GameState: setState({ phase: 'SURVIVE' or 'BANG' })
6. Effects:
   - SoundManager.play('click' or 'bang')
   - AnimationManager.playSurvive() or playBang()
7. (연출 완료 대기)
8. GameState: setState({ phase: 'PLAYING' or 'RESULT' })
9. UI: 화면 갱신
```

## 8. 연출 시스템 아키텍처

### 8.1 SoundManager

```javascript
// effects/SoundManager.js
class SoundManager {
  #sounds = {};
  #muted = false;

  async init() {
    const soundFiles = {
      spin: 'assets/sounds/spin.mp3',
      click: 'assets/sounds/click.mp3',
      bang: 'assets/sounds/bang.mp3',
      heartbeat: 'assets/sounds/heartbeat.mp3',
    };
    for (const [name, path] of Object.entries(soundFiles)) {
      this.#sounds[name] = new Audio(path);
      this.#sounds[name].preload = 'auto';
    }
  }

  play(name) {
    if (this.#muted || !this.#sounds[name]) return;
    this.#sounds[name].currentTime = 0;
    this.#sounds[name].play().catch(() => {});
  }

  setMute(muted) {
    this.#muted = muted;
  }
}
```

```javascript
// effects/AnimationManager.js
class AnimationManager {
  /** 탄창 스핀 연출 */
  async playSpinAnimation(chamberElement) {
    chamberElement.classList.add('spinning');
    await this.#wait(2000);
    chamberElement.classList.remove('spinning');
  }

  /** 생존 연출 */
  async playSurviveEffect() {
    document.body.classList.add('flash-green');
    await this.#wait(500);
    document.body.classList.remove('flash-green');
  }

  /** 꽝 연출 */
  async playBangEffect() {
    document.body.classList.add('flash-red', 'screen-shake');
    await this.#wait(800);
    document.body.classList.remove('flash-red', 'screen-shake');
  }

  #wait(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

## 10. 보안 설계

### 10.1 RNG 보안
- `crypto.getRandomValues()` 사용으로 예측 불가능한 난수 생성
- `Math.random()` 전면 금지
- 탄환 위치 정보는 `#private` 필드로 캡슐화, DOM/콘솔 비노출

### 10.2 데이터 보안
- 서버 통신 없음 → 네트워크 공격 벡터 없음
- 로컬 스토리지에 민감 정보 미저장 (이름만 저장)
- XSS 방지: 사용자 입력값 DOM 삽입 시 `textContent` 사용 (innerHTML 금지)

### 10.3 입력 검증

```javascript
// 참가자 이름 검증
function validatePlayerName(name) {
  const trimmed = name.trim();
  if (trimmed.length === 0 || trimmed.length > 10) return false;
  // 허용: 한글, 영문, 숫자, 공백
  return /^[가-힣a-zA-Z0-9\s]+$/.test(trimmed);
}
```

```
     │
     ├── src/ 폴더 → GitHub Pages로 직접 배포
     │
     ▼
https://{username}.github.io/Roulette/
```

### 11.2 GitHub Actions 워크플로

```yaml
name: Deploy to GitHub Pages
on:
  push:
    branches: [main]
    paths: ['src/**']

permissions:
  contents: read
  pages: write
  id-token: write

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/configure-pages@v4
      - uses: actions/upload-pages-artifact@v3
        with:
          path: src
      - uses: actions/deploy-pages@v4
```

## 12. 성능 최적화

### 12.1 로딩 성능

| 항목 | 목표 | 전략 |
|------|------|------|
| 첫 로딩 | < 2초 | 최소 번들, 정적 파일 |
| JS 크기 | < 50KB | Vanilla JS, 불필요 라이브러리 배제 |
| CSS 크기 | < 20KB | 최소 스타일, 유틸리티 클래스 미사용 |
| 사운드 | 지연 로드 | 게임 시작 시 로드, 작은 파일 크기 |

### 12.2 런타임 성능

| 항목 | 목표 | 전략 |
|------|------|------|
| 인터랙션 응답 | < 100ms | 이벤트 핸들러 경량화 |
| 애니메이션 | 60fps | CSS 애니메이션 / `transform` 사용 |
| DOM 조작 | 최소화 | 상태 변경 시 필요 부분만 갱신 |
| 메모리 | < 30MB | 파티클 풀링, 사운드 재사용 |

### 12.3 리소스 전략

```html
<!-- CSS: 즉시 로드 -->

<!-- JS: defer로 비차단 로드 -->
<script type="module" src="scripts/app.js" defer></script>

<!-- 사운드: 게임 시작 시 동적 로드 -->
<!-- 파티클 라이브러리: 게임 화면 진입 시 동적 import -->
```
